4) Number of applicants for jobtitle : "Accountant I'

db.applicants.aggregate([
  {
    $lookup: {
      from: "job_listings",
      localField: "JobID",
      foreignField: "JobID",
      as: "job"
    }
  },
  {
    $unwind: "$job"
  },
  {
    $match: { "job.JobTitle": "Accountant I" }
  },
  {
    $count: "TotalApplicants"
  }
])


Output :

{
  TotalApplicants: 1
}
